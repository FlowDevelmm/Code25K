public class Balanceamento {
    public static boolean estaBalanceada(String expr) {
        IPilha<Character> pilha = new PilhaVetor<>(expr.length());

        for (char c : expr.toCharArray()) {
            if (c == '(') pilha.push(c);
            else if (c == ')') {
                if (pilha.vazia()) return false;
                pilha.pop();
            }
        }
        return pilha.vazia();
    }

    public static void main(String[] args) {
        String exp = "(2*(3+4))";
        System.out.println(exp + " balanceada? " + estaBalanceada(exp));
    }
}
